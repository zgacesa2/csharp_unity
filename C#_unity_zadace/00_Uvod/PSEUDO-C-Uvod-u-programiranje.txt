        if (dead)
        {
            // if champion is dead, count up the respawn counter towards the total respawn time one per second
            // when respawnCounter equals total respawn time, player is respawned
            // we count up to avoid changing the respawn timer variable itself
            respawnCounter++;
            if (respawnCounter >= totalRespawnTime)
            {
                respawnCounter = 0;
                Spawn();
            }
            Debug.Log("Respawning in: " + (totalRespawnTime - respawnCounter) + " seconds.");
        }
        else
        {
            // if champion is alive, regen health and mana per second
            currentHealth += healthRegen;
            currentMana += manaRegen;
        }
    }

8.) Kada Perica ugleda prirodan broj, on odmah provjeri je li taj broj „nogometan“ ili kako se još kaže „najvažniji sporedan broj na svijetu“. Broj je nogometan ako je u njemu moguce pronaci tri uzastopne znamenke ciji je zbroj jednak deset. Npr. brojevi 442 i 352 su nogometni. Napiši program koji ce za zadani prirodan broj N provjeriti jel on nogometan. (unos je izmedu 100 i 999)

public int number = 442;

private void Start() {
    var sum = 0;

    if (number < 99 || number > 999) {
        // validation
        Debug.Log("Broj nije izmedu 100 i 999");
    } else {
        // number is valid
        while (number > 0) {
            var digit = number % 10; // 442 % 10 = 2
            sum += digit;

            number /= 10; // iteration -> 442 / 10 = 44....
        }

        if (sum == 10) {
            Debug.Log("Nogometan");
        } else {
            Debug.Log("Nije Nogometan");
        }
    }
}
